metronome()

TODO: Summary https://github.com/overtone/overtone/blob/f6d414f884f1b6d3166195b49276174efddf2cf2/src/overtone/music/rhythm.clj

const context = new AudioContext()

const metro = metronome(context, 60)

metro.on('tick', tick => {
  console.log('tick', tick)
})

metro.on('tick/16', tick => {
  console.log('tick/16', tick)
})

metro.start()

  s.connect(context.destination)
  s.start()
})()

sampler()

TODO: Summary

gen.run(async context => {
  const piano = await gen.sampler(
    context,
    gen.sampleMap('https://unpkg.com/@meleyal/gen/samples/piano/')
  )

  // Single C note
  piano('C4', { volume: 0.5 })

  // C major chord
  piano(['C4', 'E4', 'G4'], { volume: 0.8, duration: 2 })
})

sampler

API based on Tone.js sampler.

const context = new AudioContext()

const load = url => {
  window
    .fetch(url)
    .then(response => response.arrayBuffer())
    .then(arrayBuffer => context.decodeAudioData(arrayBuffer))
}

const play = audioBuffer => {
  const source = context.createBufferSource()
  source.buffer = audioBuffer
  source.connect(context.destination)
  source.start()
}

const sampler = samples => {
  buffered = {}
  samples.forEach((k, v) => {
    buffered[k] = fetch(v)
  })

  return note => {
    // how to play back the sample?
    play(buffered[note])
  }
}

const piano = sampler({
  c4: 'http://localhost:3000/jem/docs/assets/samples/a4.wav',
  c8: 'http://localhost:3000/jem/docs/assets/samples/c8.wav'
})
